const usuarios = [
    { usuario: "admin", password: "1234" },
    { usuario: "usuario", password: "abcd" }
];

const form = document.getElementById("loginForm");
const mensaje = document.getElementById("message");
const themeBtn = document.getElementById("themeBtn");

// Validación del Login
form.addEventListener("submit", function(e){
    e.preventDefault();

    const user = document.getElementById("username").value;
    const pass = document.getElementById("password").value;

    const valido = usuarios.find(
        u => u.usuario === user && u.password === pass
    );

    if(valido){
        mensaje.textContent = "¡Inicio de sesión correcto! 🎉";
        mensaje.style.color = "#48bb78"; // Verde
    } else {
        mensaje.textContent = "Usuario o contraseña incorrectos ❌";
        mensaje.style.color = "#f56565"; // Rojo
    }
});

// Función para actualizar el texto del botón de tema
function actualizarTextoBoton() {
    if (document.body.classList.contains("dark")) {
        themeBtn.textContent = "☀️ Modo Día";
    } else {
        themeBtn.textContent = "🌙 Modo Noche";
    }
}

// Evento Click Modo Noche
themeBtn.addEventListener("click", ()=>{
    document.body.classList.toggle("dark");
    actualizarTextoBoton();

    localStorage.setItem(
        "theme",
        document.body.classList.contains("dark") ? "dark" : "light"
    );
});

// Cargar tema guardado al iniciar
if(localStorage.getItem("theme") === "dark"){
    document.body.classList.add("dark");
    actualizarTextoBoton();
}
